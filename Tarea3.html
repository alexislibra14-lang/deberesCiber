Sentencia if

La sentencia if en Python permite ejecutar bloques de código condicionalmente si una expresión es verdadera (True). Se estructura usando if para la condición principal, elif (else if) para condiciones adicionales, y else para el caso predeterminado, siempre terminando en dos puntos (:) y usando indentación obligatoria

vocal_check.py 
# Solicitar una letra al usuario
letra = input("Introduce una letra: ").lower()

# Verificar si la letra es una vocal
if letra in "aeiou":
    print(f"La letra {letra} es una vocal")
else:
    print(f"La letra {letra} no es una vocal")
notas.py 
# Solicitar la nota al usuario
nota = int(input("Introduce la nota (0-10): "))

# Verificar aprobación (≥ 7)
if nota >= 7:
    print("Estado: APROBADO")
else:
    print("Estado: REPROBADO")

multiplo_3.py 
# Verificar múltiplo de 3
numero = int(input("Introduce un número: "))

if numero % 3 == 0:
    print(f"{numero} es múltiplo de 3")
else:
    print(f"{numero} NO es múltiplo de 3")
signo_num.py 
# Determinar Positivo, Negativo o Cero
n = float(input("Introduce número: "))

if n > 0:
    print("Es Positivo")
elif n < 0:
    print("Es Negativo")
else:
    print("Es Cero")
