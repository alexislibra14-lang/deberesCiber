Funciones

Las funciones en Python son bloques de código reutilizables, definidos con la palabra clave def, que ejecutan tareas específicas y ayudan a modularizar programas.

#Crea una función que reciba un número y determine si es par o impar. 

# 1. Definimos la función 'verificar_paridad' que recibe un parámetro llamado 'numero'
def verificar(numero):
    
    # 2. Usamos el operador '%' (módulo) para obtener el residuo de dividir entre 2.
    # Si el residuo es exactamente igual a 0, ejecutamos el primer bloque.
    if numero % 2 == 0:
        # 3. Si la condición se cumple, el número es par.
        print(f"El número {numero} es Par.")
        
    # 4. Si la condición de arriba NO se cumple (el residuo no es 0), usamos 'else'.
    else:
        # 5. Si el residuo es 1 (o cualquier otro), el número es impar.
        print(f"El número {numero} es Impar.")

# --- Bloque para probar la función ---

# 6. Pedimos un número al usuario y lo convertimos a entero con 'int()'
num_usuario = int(input("Introduce un número entero: "))

# 7. Llamamos a la función enviándole el número que el usuario escribió
verificar(num_usuario)
#Crea una función que reciba una palabra y retorne la cadena invertida. 

# 1. Definimos la función 'invertir_cadena' que recibe el parámetro 'texto'
def cadena(texto):
   
    
    # 2. Usamos la técnica de 'slicing' [inicio:fin:paso].
    # Al dejar vacíos el inicio y el fin, toma toda la cadena.
    # El '-1' en el paso indica que recorra la cadena de atrás hacia adelante.
    resultado = texto[::-1]
    
    # 3. Retornamos la cadena ya invertida
    return resultado

# --- Bloque principal del programa ---

# 4. Solicitamos al usuario que ingrese una palabra
palabra_original = input("Escribe una palabra ")

# 5. Llamamos a la función y guardamos lo que retorna en 'palabra_invertida'
palabra_invertida = cadena(palabra_original)

# 6. Mostramos ambos resultados para comparar
print(f"Original: {palabra_original}")
print(f"Invertido: {palabra_invertida}")

#Crea una función que reciba una palabra y retorne su longitud
# 1. Definimos la función 'contar_caracteres' que acepta un parámetro llamado 'palabra'
def contar(palabra):
    
    # 2. Usamos la función interna len() para obtener la longitud
    longitud = len(palabra)
    
    # 3. Retornamos el valor para que pueda ser usado fuera de la función
    return longitud

# --- Bloque principal del programa ---

# 4. Solicitamos al usuario que escriba una palabra
entrada = input("Escribe una palabra o frase: ")

# 5. Llamamos a la función y guardamos el resultado en la variable 'resultado'
resultado = contar(entrada)

# 6. Mostramos el mensaje final al usuario
print(f"La palabra '{entrada}' tiene {resultado} caracteres.")

# Crea una función que convierta una temperatura de Celsius a Fahrenheit 


# 1. Definimos la función 'convertir_celsius_a_fahrenheit' con el parámetro 'celsius'
def convertir(celsius):
  
    
    # 2. Aplicamos la fórmula matemática: (Celsius * 1.8) + 32
    fahrenheit = (celsius * 1.8) + 32
    
    # 3. Retornamos el resultado para que pueda ser usado fuera de la función
    return fahrenheit

# --- Bloque principal del programa ---

# 4. Solicitamos al usuario la temperatura (usamos float para permitir decimales)
temp_c = float(input("Introduce la temperatura en grados Celsius: "))

# 5. Llamamos a la función y guardamos el valor retornado en 'resultado'
resultado = convertir(temp_c)

# 6. Mostramos el resultado final con un formato claro
print(f"{temp_c}°C equivalen a {resultado}°F"))
#Crea una función que retorne True si un número es positivo, o False si no lo es. 

# 1. Definimos la función 'es_positivo' que recibe un parámetro llamado 'numero'
def es_positivo(numero):
   
    
    # 2. Usamos una estructura if/else para evaluar la condición
    if numero > 0:
        # 3. Si el número es mayor que 0, retornamos el valor booleano True
        return True
    else:
        # 4. En cualquier otro caso (0 o números negativos), retornamos False
        return False

# --- Bloque principal del programa ---

# 5. Pedimos un número al usuario y lo convertimos a float para mayor flexibilidad
valor = float(input("Introduce un número para verificar si es positivo: "))

# 6. Llamamos a la función y guardamos el resultado (True o False) en 'resultado'
resultado = es_positivo(valor)

# 7. Usamos el resultado para mostrar un mensaje personalizado
if resultado:
    print(f"✅ El número {valor} es positivo.")
else:
    print(f"❌ El número {valor} no es positivo (es cero o negativo).")

# Calculadora básica
def menu():
    print("Calculadora básica\n1... para sumar\n2... para restar\n3... para multiplicar\n4... para dividir")

def opcion():
    opcion = int(input("Ingrese una opción entre 1 y 4: "))
    return opcion

def ingresar_numeros():
    a = float(input("Ingrese el primer número: "))
    b = float(input("Ingrese el segundo número: "))
    return a, b

def sumar(a, b):
    print("La suma es: ", a + b)

def restar(a, b):
    print("La resta es: ", a - b)

def multiplicar(a, b):
    print("La multiplicación es: ", a * b)

def dividir(a, b):
    if b != 0:
        print("La división es: ", a / b)
    else:
        print("Error: división por cero")

def calcular(opcionSeleccionada):
    a, b = ingresar_numeros()
    if opcionSeleccionada == 1:
        sumar(a, b)
    elif opcionSeleccionada == 2:
        restar(a, b)
    elif opcionSeleccionada == 3:
        multiplicar(a, b)
    elif opcionSeleccionada == 4:
        dividir(a, b)
    else:
        print("Error: opción fuera de rango")

menu()
opcion = opcion()
calcular(opcion)
