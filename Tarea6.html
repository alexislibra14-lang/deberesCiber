import tkinter as tk
from tkinter import messagebox

def calcular_imc():
    try:
        peso = float(entry_peso.get()) 
        altura = float(entry_altura.get())
        imc = peso / (altura ** 2)
        
        if imc < 18.5:
            categoria, color = "Bajo peso", "#3498db"
        elif 18.5 <= imc < 24.9:
            categoria, color = "Peso normal", "#2ecc71"
        elif 25 <= imc < 29.9:
            categoria, color = "Sobrepeso", "#f1c40f"
        else:
            categoria, color = "Obesidad", "#e74c3c"
            
        label_resultado.config(text=f"IMC: {imc:.2f}", fg=color)
        label_info.config(text=f"Categoría: {categoria}")
        
    except ValueError:
        messagebox.showerror("Error", "Ingresa valores numéricos.")
    except ZeroDivisionError:
        messagebox.showerror("Error", "La altura no puede ser cero.")

# Ventana Principal
ventana = tk.Tk()
ventana.title("Calculadora IMC")
ventana.geometry("300x350")

tk.Label(ventana, text="Calculadora de IMC", font=("Arial", 14, "bold")).pack(pady=10)
tk.Label(ventana, text="Peso (kg):").pack()
entry_peso = tk.Entry(ventana, justify="center")
entry_peso.pack()

tk.Label(ventana, text="Altura (m):").pack(pady=5)
entry_altura = tk.Entry(ventana, justify="center")
entry_altura.pack()

tk.Button(ventana, text="Calcular", command=calcular_imc, bg="#2c3e50", fg="white").pack(pady=20)

label_resultado = tk.Label(ventana, text="IMC: --", font=("Arial", 12, "bold"))
label_resultado.pack()

label_info = tk.Label(ventana, text="Esperando datos...", font=("Arial", 10))
label_info.pack()

ventana.mainloop()
