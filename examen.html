import tkinter as tk
from tkinter import messagebox

def calcular_gastos():
    try:
        ali = float(entry_comida.get()) 
        tra = float(entry_transporte.get())
        viv = float(entry_vivienda.get())
        total = ali + tra + viv
        
        color = "#e74c3c" if total > 500 else "#2ecc71"
        res = "Elevado" if total > 500 else "Saludable"
            
        label_resultado.config(text=f"${total:.2f}", fg=color)
        label_info.config(text=f"Estado: {res}")
        
    except ValueError:
        messagebox.showerror("Error", "Dato no válido")

ventana = tk.Tk()
ventana.title("EduGastos")
ventana.geometry("300x400")

tk.Label(ventana, text="Gastos Mensuales", font=("Arial", 12, "bold")).pack(pady=10)
# ... campos de entrada ...
ventana.mainloop()
